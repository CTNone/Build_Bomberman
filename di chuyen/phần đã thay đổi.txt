scene.setOnKeyPressed(e -> {

            switch (e.getCode()) {
                case UP:
                    bomberman.moveUp();
                    break;
                case DOWN:
                    bomberman.moveDown();
                    break;
                case LEFT:
                    bomberman.moveLeft();
                    break;
                case RIGHT:
                    bomberman.moveRight();
                    break;
                default:
                    break;
            }
        });

// Bomber : thêm vào
